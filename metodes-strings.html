Apunts de Mètodes de Strings

trim(): Elimina los espacios en blanco al principio y al final de una cadena.

             // Apunts de Mètodes de Strings
 
             // trim(): Elimina los espacios en blanco al principio y al final de una cadena.
             let text = "   Hello World   ";
             console.log(text.trim()); // "Hello World"
             
             // padStart() y padEnd(): Añade caracteres al inicio o al final de una cadena hasta alcanzar la longitud deseada.
             let text = "Hello";
             console.log(text.padStart(10, '*')); // "*****Hello"
             console.log(text.padEnd(10, '*'));   // "Hello*****"
             
             // replace() y replaceAll(): Reemplaza una subcadena por otra.
             // replace() solo reemplaza la primera aparición, mientras que replaceAll() reemplaza todas las apariciones.
             let text = "Hello World";
             console.log(text.replace("World", "Universe")); // "Hello Universe"
             console.log(text.replaceAll("l", "L")); // "HeLLo WorLd"
             
             // split(): Divide una cadena en un array de subcadenas basadas en un delimitador.
             let text = "Hello World";
             console.log(text.split(" ")); // ["Hello", "World"]
             
             // toUpperCase() y toLowerCase(): Convierte una cadena a mayúsculas o minúsculas.
             let text = "Hello World";
             console.log(text.toUpperCase()); // "HELLO WORLD"
             console.log(text.toLowerCase()); // "hello world"
             
             // slice(): Extrae una parte de una cadena sin modificar la original.
             let text = "Hello World";
             console.log(text.slice(0, 5)); // "Hello"
             
             // Mètodes de Búsqueda
             
             // indexOf(): Devuelve el índice de la primera aparición de una subcadena.
             let text = "Hello World";
             console.log(text.indexOf("World")); // 6
             
             // includes(): Verifica si una subcadena está incluida en la cadena.
             let text = "Hello World";
             console.log(text.includes("World")); // true
             
             // startsWith() y endsWith(): Verifican si la cadena empieza o termina con una subcadena específica.
             let text = "Hello World";
             console.log(text.startsWith("Hello")); // true
             console.log(text.endsWith("World")); // true
             
             // Gestió de Dades
             
             // Gestionar dades amb bucles i condicionals.
             
             let data = "NOM Omar COG Olmedo AGE 33";
             
             // Dividir la cadena de datos en un array de palabras
             let dataArray = data.split(" ");
             
             // Crear un objeto para almacenar los datos
             let dataObject = {};
             
             // Recorrer el array y asignar los valores al objeto
             for (let i = 0; i < dataArray.length; i += 2) {
                 let key = dataArray[i];
                 let value = dataArray[i + 1];
                 dataObject[key] = value;
             }
             
             // Imprimir los datos del objeto en formato legible
             console.log("Datos del objeto:");
             for (let key in dataObject) {
                 if (dataObject.hasOwnProperty(key)) {
                     console.log(`${key}: ${dataObject[key]}`);
                 }
             }

             if (parseInt(dataObject.AGE) > 30) { // Convertir la edad a número para la comparación
